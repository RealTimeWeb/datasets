014/**
015 * 
016 */
017public class Daily {
018        
019    // The amount of money that the publisher made per day on this book, in dollars.
020    private Double publisherRevenue;
021    // The amount of money that Amazon made per day on this book, in dollars.
022    private Double amazonRevenue;
023    // The amount of money that the author made per day on this book, in dollars.
024    private Double authorRevenue;
025    // The number of books sold per day.
026    private Integer unitsSold;
027    // The total amount of money that was made per day on this book, in dollars.
028    private Double grossSales;
029    
030    
031    /*
032     * @return 
033     */
034    public Double getPublisherRevenue() {
035        return this.publisherRevenue;
036    }
037    
038    
039    
040    /*
041     * @return 
042     */
043    public Double getAmazonRevenue() {
044        return this.amazonRevenue;
045    }
046    
047    
048    
049    /*
050     * @return 
051     */
052    public Double getAuthorRevenue() {
053        return this.authorRevenue;
054    }
055    
056    
057    
058    /*
059     * @return 
060     */
061    public Integer getUnitsSold() {
062        return this.unitsSold;
063    }
064    
065    
066    
067    /*
068     * @return 
069     */
070    public Double getGrossSales() {
071        return this.grossSales;
072    }
073    
074    
075    
076        
077        /**
078         * Creates a string based representation of this Daily.
079        
080         * @return String
081         */
082        public String toString() {
083                return "Daily[" +publisherRevenue+", "+amazonRevenue+", "+authorRevenue+", "+unitsSold+", "+grossSales+"]";
084        }
085        
086        /**
087         * Internal constructor to create a Daily from a  representation.
088         * @param json_data The raw json data that will be parsed.
089         */
090    public Daily(JSONObject json_data) {
091        try {// publisher revenue
092            this.publisherRevenue = ((Number)json_data.get("publisher revenue")).doubleValue();// amazon revenue
093            this.amazonRevenue = ((Number)json_data.get("amazon revenue")).doubleValue();// author revenue
094            this.authorRevenue = ((Number)json_data.get("author revenue")).doubleValue();// units sold
095            this.unitsSold = ((Number)json_data.get("units sold")).intValue();// gross sales
096            this.grossSales = ((Number)json_data.get("gross sales")).doubleValue();
097        } catch (NullPointerException e) {
098                System.err.println("Could not convert the response to a Daily; a field was missing.");
099                e.printStackTrace();
100        } catch (ClassCastException e) {
101                System.err.println("Could not convert the response to a Daily; a field had the wrong structure.");
102                e.printStackTrace();
103        }
104        }       
105}
