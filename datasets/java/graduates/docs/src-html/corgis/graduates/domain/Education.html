018public class Education {

024    
025    /**
026     * The name of the major for these graduated students.
027     * @return String

029    public String getMajor() {

031    }

033    
034    
035    /**
036     *

039    public Degrees getDegrees() {

043    
044    
045        
046        /**
047         * Creates a string based representation of this Education.
048        
049         * @return String
050         */
051        public String toString() {
052                return "Education[" +major+", "+degrees+"]";
053        }
054        
055        /**
056         * Internal constructor to create a Education from a  representation.
057         * @param json_data The raw json data that will be parsed.
058         */
059    public Education(JSONObject json_data) {
060        //System.out.println(json_data);
061        
062        try {
063            // Major
064            this.major = (String)json_data.get("Major");
065        } catch (NullPointerException e) {
066                System.err.println("Could not convert the response to a Education; the field major was missing.");
067                e.printStackTrace();
068        } catch (ClassCastException e) {
069                System.err.println("Could not convert the response to a Education; the field major had the wrong structure.");
070                e.printStackTrace();
071        }
072        
073        try {
074            // Degrees
075            this.degrees = new Degrees((JSONObject)json_data.get("Degrees"));
076        } catch (NullPointerException e) {
077                System.err.println("Could not convert the response to a Education; the field degrees was missing.");
078                e.printStackTrace();
079        } catch (ClassCastException e) {
080                System.err.println("Could not convert the response to a Education; the field degrees had the wrong structure.");
081                e.printStackTrace();
082        }
083        
084        }
