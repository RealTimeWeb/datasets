045        /**

047

052        }
053        
054        /**
055         * Internal constructor to create a Birth from a  representation.
056         * @param json_data The raw json data that will be parsed.
057         */
058    public Birth(JSONObject json_data) {
059        //System.out.println(json_data);
060        
061        try {
062            // date
063            this.date = new Date((JSONObject)json_data.get("date"));
064        } catch (NullPointerException e) {
065                System.err.println("Could not convert the response to a Birth; the field date was missing.");
066                e.printStackTrace();
067        } catch (ClassCastException e) {
068                System.err.println("Could not convert the response to a Birth; the field date had the wrong structure.");
069                e.printStackTrace();
070        }
071        
072        try {
073            // location name
074            this.locationName = (String)json_data.get("location name");
075        } catch (NullPointerException e) {
076                System.err.println("Could not convert the response to a Birth; the field locationName was missing.");
077                e.printStackTrace();
078        } catch (ClassCastException e) {
079                System.err.println("Could not convert the response to a Birth; the field locationName had the wrong structure.");
080                e.printStackTrace();
081        }

083        }
